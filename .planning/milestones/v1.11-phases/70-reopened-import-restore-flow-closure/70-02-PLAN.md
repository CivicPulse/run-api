test("reopened import proceeds through confirm to progress step", async ({

  await setupAuth(page)

  // Track confirm call count to switch job status after confirm
  let confirmed = false

  await page.route("**/api/v1/**", (route) => {
    const url = route.request().url()
    const method = route.request().method()

    // Reuse all standard mocks from setupApiMocks pattern...
    // (copy the standard mock responses for /config/public, /me, /me/campaigns, /me/orgs, campaign detail, voter imports list)

    // Confirm mapping — POST /imports/{jobId}/confirm
    if (url.includes(`/imports/${JOB_ID}/confirm`) && method === "POST") {
      confirmed = true
      return route.fulfill({
        status: 200,
        contentType: "application/json",
        body: JSON.stringify({ id: JOB_ID, status: "queued" }),
      })
    }

    // Import job detail — status depends on whether confirm happened
    if (url.includes(`/imports/${JOB_ID}`) && method === "GET" && !url.includes("/detect")) {
      return route.fulfill({
        status: 200,
        contentType: "application/json",
        body: JSON.stringify({
          id: JOB_ID,
          campaign_id: CAMPAIGN_ID,
          original_filename: "example-2026-02-24.csv",
          status: confirmed ? "processing" : "uploaded",
          detected_columns: L2_COLUMNS,
          suggested_mapping: L2_MAPPING,
          format_detected: "l2",
