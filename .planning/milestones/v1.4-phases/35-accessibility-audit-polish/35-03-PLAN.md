    Create `web/e2e/phase35-touch-targets.spec.ts` — a Playwright e2e test that verifies all interactive elements in field routes meet WCAG 2.5.5 44x44px minimum.

    Use the established `page.route()` API mocking pattern from prior phases. The test must:

    1. Mock API responses for `/api/campaigns/*/field/me`, walk list entries, and call list entries to render the canvassing and phone banking screens with real-looking data.

    2. For each field route (canvassing, phone-banking), navigate to the route and scan all visible interactive elements using the selector: `'button, a[href], [role="button"], input, [role="tab"]'`.

    3. For each visible element, get its `boundingBox()` and assert both width and height are >= 44. Skip elements where `isVisible()` returns false or `boundingBox()` returns null.

    4. Include descriptive failure messages that identify which element failed: use `await el.getAttribute('aria-label')` or `await el.textContent()` to identify the element in the error message.

    Structure:
    ```typescript
    import { test, expect } from "@playwright/test"

    // Mock data for field/me, walk list entries, call list entries
    // ... (use patterns from existing e2e tests)

    async function assertTouchTargets(page: Page, route: string) {
      const interactives = page.locator(
        'button, a[href], [role="button"], input, [role="tab"]'
      )
      const count = await interactives.count()
      expect(count).toBeGreaterThan(0) // Sanity check

      const violations: string[] = []
      for (let i = 0; i < count; i++) {
        const el = interactives.nth(i)
        if (!(await el.isVisible())) continue
        const box = await el.boundingBox()
        if (!box) continue

        if (box.width < 44 || box.height < 44) {
          const label = await el.getAttribute("aria-label") || await el.textContent() || `element-${i}`
          violations.push(
            `"${label.trim().slice(0, 50)}" is ${Math.round(box.width)}x${Math.round(box.height)}px (min 44x44)`
          )
        }
      }
      expect(violations, `Touch target violations on ${route}`).toEqual([])
    }

    test.describe("WCAG 2.5.5 Touch Targets", () => {
      test.beforeEach(async ({ page }) => {
        // Mock all field API endpoints with page.route()
      })

      test("canvassing route: all interactive elements >= 44x44px", async ({ page }) => {
        await page.goto("/field/test-campaign/canvassing")
        await assertTouchTargets(page, "canvassing")
      })

      test("phone-banking route: all interactive elements >= 44x44px", async ({ page }) => {
        await page.goto("/field/test-campaign/phone-banking")
        await assertTouchTargets(page, "phone-banking")
      })
    })
    ```

    - Walk list entries with 2-3 voters at one household
    - Call list entries with 2-3 entries
    - Walk list detail with a script_id
    - Survey script with 1 question (for InlineSurvey if opened)

    The test should run independently without a live backend.
