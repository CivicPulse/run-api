    ```python
    """Integration test: full L2 import flow with sample file.

    Per D-19: Verifies suggest_field_mapping + parse_voting_history + apply_field_mapping
    work correctly together with the real L2 sample CSV.
    """
    from __future__ import annotations

    import csv
    from pathlib import Path

    import pytest

    from app.services.import_service import (
        suggest_field_mapping,
        detect_l2_format,
        parse_voting_history,
        apply_field_mapping,
    )

    _SAMPLE_PATH = Path(__file__).resolve().parents[2] / "data" / "example-2026-02-24.csv"


    def _read_sample_rows(max_rows: int = 5) -> tuple[list[str], list[dict[str, str]]]:
        """Read headers and first N data rows from L2 sample."""
        with open(_SAMPLE_PATH) as f:
            reader = csv.DictReader(f)
            headers = reader.fieldnames or []
            rows = []
            for i, row in enumerate(reader):
                if i >= max_rows:
                    break
                rows.append(row)
        return list(headers), rows


    @pytest.mark.integration
    class TestL2FullImportFlow:
        def test_l2_format_detected(self):
            """L2 sample file is detected as L2 format."""
            headers, _ = _read_sample_rows(0)
            mapping = suggest_field_mapping(headers)
            fmt = detect_l2_format(mapping)
            assert fmt == "l2", f"Expected 'l2', got '{fmt}'"

        def test_all_data_columns_mapped(self):
            """All 47 non-history columns map to a canonical field."""
            headers, _ = _read_sample_rows(0)
            mapping = suggest_field_mapping(headers)

            history_cols = {
                "General_2024", "Primary_2024",
                "Voted in 2022", "Voted in 2020", "Voted in 2018",
                "Voted in 2022 Primary", "Voter in 2020 Primary", "Voted in 2018 Primary",
            }
            unmapped = []
            for col in headers:
                if col in history_cols:
                    continue
                entry = mapping.get(col, {})
                if entry.get("field") is None:
                    unmapped.append(col)
            assert unmapped == [], f"Unmapped data columns: {unmapped}"

        def test_voting_history_parsed_for_all_rows(self):
            """Each sample row's voting history columns parse into canonical entries."""
            _, rows = _read_sample_rows(5)
            for i, row in enumerate(rows):
                history = parse_voting_history(row)
                # Every row should have some voting history (sample has Y/A/E values)
                # At minimum, check that results are canonical format
                for entry in history:
                    assert entry.startswith("General_") or entry.startswith("Primary_"), (
                        f"Row {i}: non-canonical history entry '{entry}'"
                    )

        def test_apply_field_mapping_populates_new_columns(self):
            """apply_field_mapping correctly maps L2 CSV values to canonical field names including new columns."""
            headers, rows = _read_sample_rows(3)
            mapping_result = suggest_field_mapping(headers)

            # Build the mapping dict: csv_col -> canonical_field (only mapped columns)
            field_mapping = {}
            for col, entry in mapping_result.items():
                if entry.get("field") is not None:
                    field_mapping[col] = entry["field"]

            for i, row in enumerate(rows):
                mapped = apply_field_mapping(row, field_mapping)

                # Verify core fields populated
                assert mapped.get("first_name"), f"Row {i}: first_name missing"
                assert mapped.get("last_name"), f"Row {i}: last_name missing"
                assert mapped.get("source_id"), f"Row {i}: source_id missing"

                # Verify new L2 columns populated (if CSV has values)
                # house_number should exist for rows with address data
                if row.get("House Number"):
                    assert mapped.get("house_number") == row["House Number"], (
                        f"Row {i}: house_number mismatch"
                    )

                if row.get("Mailng Designator"):
                    assert mapped.get("mailing_designator") == row["Mailng Designator"], (
                        f"Row {i}: mailing_designator mismatch (typo alias)"
                    )

                if row.get("Mailing Aptartment Number"):
                    assert mapped.get("mailing_apartment_number") == row["Mailing Aptartment Number"], (
                        f"Row {i}: mailing_apartment_number mismatch (typo alias)"
                    )

        def test_integer_coercion_mailing_household_size(self):
            """mailing_household_size CSV string values coerce to integers."""
            headers, rows = _read_sample_rows(3)
            mapping_result = suggest_field_mapping(headers)

            field_mapping = {}
            for col, entry in mapping_result.items():
                if entry.get("field") is not None:
                    field_mapping[col] = entry["field"]

            for i, row in enumerate(rows):
                mapped = apply_field_mapping(row, field_mapping)
                val = mapped.get("mailing_household_size")
                if val is not None:
                    assert isinstance(val, int), (
                        f"Row {i}: mailing_household_size should be int, got {type(val).__name__}: {val}"
                    )
    ```

    Note: If `apply_field_mapping` is not a standalone function (it may be part of a larger import pipeline), adjust the test to call whatever the correct entry point is for mapping CSV row data to canonical field names. Read the function signatures in import_service.py to determine the right approach.
