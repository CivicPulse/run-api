```ts
import { useQuery } from "@tanstack/react-query"
import { api } from "@/api/client"

export interface CallerCheckInStatus {
  id: string
  session_id: string
  user_id: string
  check_in_at: string | null
  check_out_at: string | null
  checked_in: boolean
}

export function useCallerCheckInStatus(campaignId: string, sessionId: string) {
  const query = useQuery({
    queryKey: ["caller-check-in", campaignId, sessionId],
    queryFn: async () => {
      const res = await api
        .get(`campaigns/${campaignId}/phone-bank-sessions/${sessionId}/callers/me`, {
          throwHttpErrors: false,
        })
      if (res.status === 404) return null
      if (!res.ok) throw new Error(`Unexpected status ${res.status}`)
      return res.json<CallerCheckInStatus>()
    },
    retry: (failureCount, _err) => failureCount < 2,
    staleTime: 30_000,
  })
  return {
    data: query.data,
    isLoading: query.isLoading,
    isError: query.isError,
    notAssigned: query.data === null,
  }
}
```

Adjust ky options based on the actual api client conventions in web/src/api/client.ts. If `throwHttpErrors: false` isn't supported by this wrapper, use a try/catch with ky HTTPError and inspect `error.response.status === 404`.
