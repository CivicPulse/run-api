
Target:
```ts
try { return await api.get("api/v1/org/campaigns").json<OrgCampaign[]>() }

  const is404 = err instanceof HTTPError && err.response.status === 404
  if (err instanceof PermissionError || is404) {
    return fallbackFromCampaignsList()  // existing fallback code
  }
  throw err  // let 401/500/network errors propagate
}
```
