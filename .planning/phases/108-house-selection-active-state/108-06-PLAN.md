import { test, expect } from "@playwright/test"

// Reuse the mock fixture and navigation helpers from canvassing-wizard.spec.ts.
// If those helpers are not exported, either (a) extract them to a shared
// e2e/helpers/canvassing-fixture.ts module in a separate sub-task, or (b)
// copy the relevant setup verbatim into this file. Prefer extraction ONLY

test.describe("Phase 108 — canvassing house selection", () => {
  test.beforeEach(async ({ page }) => {
    // Mount the mock API routes and walk list fixture (House A / B / C from
    // Plan 108-01 Wave 0). Navigate to /field/{campaign}/canvassing.
    //
    // Executor: copy the beforeEach block from canvassing-wizard.spec.ts and
    // confirm House C is present in the fixture (Plan 108-01 extended it).
  })

  test("SELECT-01: tapping a list row visibly swaps the rendered HouseholdCard", async ({ page }) => {
    // 1. Open the household list sheet (click the "List" toggle or equivalent).
    // 2. Assert the currently active card shows House A's address.
    // 3. Locate the House B row button by aria-label (`Jump to door 2,`).
    // 4. Click it.
    // 5. Assert the sheet closes (onOpenChange false from D-03).
    // 6. Assert the rendered HouseholdCard address heading now shows House B.
    //    Use getByRole('heading') or a specific data-testid — whichever the
    //    existing phase 107 tests use.
    // 7. Assert the ARIA live region announced "Now at {House B address}".

    await expect(page.getByRole("heading", { name: /House A address/ })).toBeVisible()
    await page.getByRole("button", { name: /list/i }).click()
    await page.getByRole("button", { name: /Jump to door 2/ }).click()
    await expect(page.getByRole("heading", { name: /House B address/ })).toBeVisible()
  })

  test("SELECT-02: tapping a map marker visibly swaps the HouseholdCard and pans the map", async ({ page }) => {
    // 1. Open the map toggle (or confirm map view is visible if dual-pane).
    // 2. Assert the active marker currently reflects House A (role=button,
    //    aria-pressed=true, aria-label="Activate door: House A address").
    // 3. Locate the House C marker via aria-label "Activate door: 400 Cherry St...".
    // 4. Click it.
    // 5. Assert the rendered HouseholdCard swaps to House C.
    // 6. Assert the previously-active marker is no longer aria-pressed=true.
    // 7. Assert the new active marker is aria-pressed=true.
    // 8. Capture a screenshot to web/screenshots/108-map-tap-active.png.
    //    (screenshots/ is gitignored per CLAUDE.md.)

    const houseCMarker = page.getByRole("button", { name: /Activate door: 400 Cherry St/ })
    await expect(houseCMarker).toBeVisible()
    await houseCMarker.click()

    await expect(page.getByRole("heading", { name: /400 Cherry St/ })).toBeVisible()
    await expect(houseCMarker).toHaveAttribute("aria-pressed", "true")
    await page.screenshot({ path: "screenshots/108-map-tap-active.png", fullPage: false })
  })

  test("SELECT-02: keyboard Enter on a focused map marker activates the same transition", async ({ page }) => {
    // 1. Navigate to the canvassing route and ensure markers are rendered.
    // 2. Focus the House C marker via keyboard (tab order OR page.focus via
    //    aria-label selector).
    // 3. Press Enter.
    // 4. Assert the rendered HouseholdCard swaps to House C.
    // 5. (Optional) Press Space on a different marker; assert same result.
    //    Only include the Space test if Plan 108-01 Spike A1 confirmed the
    //    keydown listener is necessary.

    const houseBMarker = page.getByRole("button", { name: /Activate door: 200 Oak/ })
    await houseBMarker.focus()
    await page.keyboard.press("Enter")
    await expect(page.getByRole("heading", { name: /200 Oak/ })).toBeVisible()
  })

  test("SELECT-03 resume: returning to the canvassing route lands on the persisted active house", async ({ page }) => {
    // 1. Start on canvassing route with active house = House A.
    // 2. Tap a list row to activate House B.
    // 3. Navigate to a different route (e.g. the walk list index or field hub).
    // 4. Navigate BACK to /field/{campaign}/canvassing.
    // 5. Assert the rendered HouseholdCard is still House B (persisted via
    //    sessionStorage).
    // 6. Assert the map marker for House B is aria-pressed=true.

    await page.getByRole("button", { name: /list/i }).click()
    await page.getByRole("button", { name: /Jump to door 2/ }).click()
    await expect(page.getByRole("heading", { name: /House B address/ })).toBeVisible()

    await page.goto("/field/camp-1/walk-lists")
    await page.goto("/field/camp-1/canvassing")

    await expect(page.getByRole("heading", { name: /House B address/ })).toBeVisible()
  })
})
