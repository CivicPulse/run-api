import { renderHook, act, waitFor } from "@testing-library/react"
import { useCanvassingWizard } from "./useCanvassingWizard"

// Reuse the same mock server / fixture shape from useCanvassingWizard.test.ts.
// If that file exports a helper, import it; otherwise copy the MSW/fetch mock
// setup verbatim. The walk list MUST have 3 households:
//   index 0 = house-a (1 voter, pending)
//   index 1 = house-b (1 voter, pending)  ← target for most tests
//   index 2 = house-c (1 voter, pending)

const CANVASSING_PERSIST_KEY = "canvassing-wizard"

// Helper: seed sessionStorage with a zustand-persist-v1 canvassingStore blob
// The exact shape MUST match canvassingStore.ts partialize output.
// Executor: read canvassingStore.ts and inline the correct shape here.
function seedResumeState(walkListId: string, currentAddressIndex: number) {
  const persistedState = {
    state: {
      walkListId,
      currentAddressIndex,
      completedEntries: [],
      skippedEntries: [],
      lastActiveAt: new Date().toISOString(),
      sortMode: "sequence",
      locationStatus: "idle",
      locationSnapshot: null,
    },
    version: 1,
  }
  sessionStorage.setItem(CANVASSING_PERSIST_KEY, JSON.stringify(persistedState))
}

function expectTargetReached(
  wizard: ReturnType<typeof useCanvassingWizard>,
  targetIndex: number,
  targetKey: string,
) {
  expect(wizard.currentAddressIndex).toBe(targetIndex)
  expect(wizard.currentHousehold?.householdKey).toBe(targetKey)
}

describe("SELECT-03 — active-house state machine reachability", () => {
  beforeEach(() => {
    sessionStorage.clear()
    // Reset the canvassing store between tests so persisted state from one
    // test does not leak into the next. Executor: use whichever reset helper
    // the existing tests use (often canvassingStore.getState().reset?.() or
    // a direct store.setState(initialState)).
  })

  it("list-tap reaches the target household (D-09 entry point 1)", async () => {
    const { result } = renderHook(() => useCanvassingWizard("camp-1", "walk-1"))
    await waitFor(() => expect(result.current.households.length).toBeGreaterThanOrEqual(3))
    expect(result.current.currentAddressIndex).toBe(0)

    act(() => {
      result.current.handleJumpToAddress(1)
    })

    expectTargetReached(result.current, 1, "house-b")
  })

  it("map-tap reaches the target household (D-09 entry point 2)", async () => {
    // Map-tap and list-tap share the same hook code path (handleJumpToAddress).
    // This test exercises the same call explicitly so the state-machine test
    // file documents both entry points per D-09. If the code path ever diverges,
    // this test is the regression guard that they stay in sync.
    const { result } = renderHook(() => useCanvassingWizard("camp-1", "walk-1"))
    await waitFor(() => expect(result.current.households.length).toBeGreaterThanOrEqual(3))

    act(() => {
      result.current.handleJumpToAddress(1)  // same call used by CanvassingMap marker click
    })

    expectTargetReached(result.current, 1, "house-b")
  })

  it("auto-advance after HOUSE_LEVEL_OUTCOME reaches the next household (D-09 entry point 3)", async () => {
    const { result } = renderHook(() => useCanvassingWizard("camp-1", "walk-1"))
    await waitFor(() => expect(result.current.households.length).toBeGreaterThanOrEqual(3))
    expect(result.current.currentAddressIndex).toBe(0)

    // Submit a house-level outcome on the current (house-a) voter. This

    await act(async () => {
      // Executor: replicate the 107-04 hook test's handleOutcome call here.

    })

    await waitFor(() => {
      expectTargetReached(result.current, 1, "house-b")
    })
  })

  it("skip reaches the next pending household (D-09 entry point 4)", async () => {
    const { result } = renderHook(() => useCanvassingWizard("camp-1", "walk-1"))
    await waitFor(() => expect(result.current.households.length).toBeGreaterThanOrEqual(3))
    expect(result.current.currentAddressIndex).toBe(0)

    await act(async () => {
      await result.current.handleSkipAddress()
    })

    await waitFor(() => {
      expectTargetReached(result.current, 1, "house-b")
