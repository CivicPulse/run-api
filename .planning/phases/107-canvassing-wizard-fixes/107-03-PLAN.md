```python
"""Integration tests locking the CANV-03 optional-notes contract for door knocks.

Phase 107 D-10/D-16: The door-knock POST must accept `notes=None`, `notes=""`,
an omitted `notes` field, and a real notes string. This test prevents future
regressions from re-introducing a NOT NULL or min-length constraint on the
notes column.
"""

from __future__ import annotations

import pytest
from httpx import AsyncClient

pytestmark = [pytest.mark.integration, pytest.mark.asyncio]


# NOTE: executor must wire these fixtures to whatever pattern already exists
# in tests/integration/. If test_walk_lists.py or test_canvass*.py already
# has `campaign`, `walk_list`, `walk_list_entry`, `voter`, and an
# authenticated `client` fixture, import/reuse them via conftest.py.


async def _post_door_knock(
    client: AsyncClient,
    *,
    campaign_id: str,
    walk_list_id: str,
    entry_id: str,
    voter_id: str,
    notes,  # str | None | "omit"
) -> "httpx.Response":
    payload: dict = {
        "voter_id": str(voter_id),
        "walk_list_entry_id": str(entry_id),
        "result_code": "not_home",
    }
    if notes != "omit":
        payload["notes"] = notes
    return await client.post(
        f"/api/v1/campaigns/{campaign_id}/walk_lists/{walk_list_id}/entries/{entry_id}/door_knocks",
        json=payload,
    )


async def test_post_door_knock_with_null_notes_returns_201(
    authenticated_client, seeded_walk_list_entry
):
    response = await _post_door_knock(
        authenticated_client,
        campaign_id=seeded_walk_list_entry.campaign_id,
        walk_list_id=seeded_walk_list_entry.walk_list_id,
        entry_id=seeded_walk_list_entry.id,
        voter_id=seeded_walk_list_entry.voter_id,
        notes=None,
    )
    assert response.status_code == 201, response.text
    assert response.json()["notes"] is None


async def test_post_door_knock_with_empty_string_notes_returns_201(
    authenticated_client, seeded_walk_list_entry
):
    response = await _post_door_knock(
        authenticated_client,
        campaign_id=seeded_walk_list_entry.campaign_id,
        walk_list_id=seeded_walk_list_entry.walk_list_id,
        entry_id=seeded_walk_list_entry.id,
        voter_id=seeded_walk_list_entry.voter_id,
        notes="",
    )
    assert response.status_code == 201, response.text
    # Observed contract: empty string may be preserved as "" or normalized to
    # None. Both are acceptable so long as no validation error is raised.
    assert response.json()["notes"] in ("", None)


async def test_post_door_knock_with_real_notes_echoes_back(
    authenticated_client, seeded_walk_list_entry
):
    response = await _post_door_knock(
        authenticated_client,
        campaign_id=seeded_walk_list_entry.campaign_id,
        walk_list_id=seeded_walk_list_entry.walk_list_id,
        entry_id=seeded_walk_list_entry.id,
        voter_id=seeded_walk_list_entry.voter_id,
        notes="Left a flyer at the door.",
    )
    assert response.status_code == 201, response.text
    assert response.json()["notes"] == "Left a flyer at the door."


async def test_post_door_knock_omitting_notes_field_returns_201(
    authenticated_client, seeded_walk_list_entry
):
    response = await _post_door_knock(
        authenticated_client,
        campaign_id=seeded_walk_list_entry.campaign_id,
        walk_list_id=seeded_walk_list_entry.walk_list_id,
        entry_id=seeded_walk_list_entry.id,
        voter_id=seeded_walk_list_entry.voter_id,
        notes="omit",
    )
    assert response.status_code == 201, response.text
```
