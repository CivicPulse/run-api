    2. For each router, iterate over `router.routes`
    3. For each route, check that the endpoint function has rate limit metadata
    4. SlowAPI stores rate limits in `limiter._rate_limits` dict keyed by endpoint function, OR the decorated function has attributes set by slowapi

    **Alternative simpler approach (source code inspection):**
    For each `.py` file in `app/api/v1/` (excluding `__init__.py`, `router.py`):
    1. Read the source code
    2. Find all `@router.(get|post|patch|put|delete)` decorators
    3. Verify the NEXT decorator line is `@limiter.limit(`
    4. For authenticated endpoints (those with `require_role` or `get_current_user`), verify `key_func=get_user_or_ip_key` is present in the limiter decorator

    ```python
    """Verify all API endpoints have rate limiting decorators (OBS-03, OBS-04)."""
    import ast
    import os
    from pathlib import Path

    ROUTE_DIR = Path(__file__).parent.parent.parent / "app" / "api" / "v1"
    SKIP_FILES = {"__init__.py", "router.py"}

    def _get_route_files():
        return sorted(
            f for f in ROUTE_DIR.glob("*.py")
            if f.name not in SKIP_FILES
        )

    def _parse_endpoints(filepath):
        """Parse a route file and return (function_name, has_limiter, has_user_key, is_authenticated) tuples."""
        source = filepath.read_text()
        tree = ast.parse(source)
        results = []
        for node in ast.walk(tree):
            if isinstance(node, ast.AsyncFunctionDef):
                # Check decorators
                has_router = False
                has_limiter = False
                has_user_key = False
                is_authenticated = False
                for dec in node.decorator_list:
                    dec_str = ast.dump(dec)
                    if "router" in ast.dump(dec) and any(
                        m in ast.dump(dec) for m in ["get", "post", "patch", "put", "delete"]
                    ):
                        has_router = True
                    if "limiter" in ast.dump(dec):
                        has_limiter = True
                        if "get_user_or_ip_key" in ast.dump(dec):
                            has_user_key = True
                # Check if authenticated (has require_role or get_current_user in params)
                for default in [arg for arg in ast.dump(node)]:
                    pass  # simplified
                source_lines = source.split("\n")
                func_source = ast.get_source_segment(source, node) or ""
                if "require_role" in func_source or "get_current_user" in func_source:
                    is_authenticated = True
                if has_router:
                    results.append((node.name, has_limiter, has_user_key, is_authenticated))
        return results

    def test_all_endpoints_have_rate_limit_decorator():
        """Every @router endpoint must have @limiter.limit decorator (OBS-03)."""
        missing = []
        for filepath in _get_route_files():
            endpoints = _parse_endpoints(filepath)
            for name, has_limiter, _, _ in endpoints:
                if not has_limiter:
                    missing.append(f"{filepath.name}::{name}")
        assert not missing, f"Endpoints missing @limiter.limit: {missing}"

    def test_authenticated_endpoints_use_user_key():
        """Authenticated endpoints must use get_user_or_ip_key (OBS-04)."""
        missing = []
        for filepath in _get_route_files():
            endpoints = _parse_endpoints(filepath)
            for name, has_limiter, has_user_key, is_authenticated in endpoints:
                if is_authenticated and has_limiter and not has_user_key:
                    missing.append(f"{filepath.name}::{name}")
        assert not missing, f"Authenticated endpoints missing get_user_or_ip_key: {missing}"
    ```
