Test 1: `"every marker image in the canvassing map has naturalWidth > 0"`:

```typescript
test("every marker image in the canvassing map has naturalWidth > 0", async ({ page }) => {
  const failedRequests: string[] = []
  page.on("requestfailed", (req) => {
    const url = req.url()
    if (/marker-|leaflet.*\.png/.test(url)) {
      failedRequests.push(url)
    }
  })

  const badStatuses: { url: string; status: number }[] = []
  page.on("response", async (res) => {
    const url = res.url()
    if (/marker-.*\.png|marker-shadow.*\.png/.test(url) && res.status() >= 400) {
      badStatuses.push({ url, status: res.status() })
    }
  })

  await page.goto("/field/<seed-campaign-id>/canvassing")
  await page.waitForSelector('[data-testid="canvassing-map-container"]')
  // Give Leaflet a tick to paint markers.
  await page.waitForFunction(() => {
    const imgs = document.querySelectorAll(".leaflet-container img")
    return imgs.length > 0
  })

  const widths = await page.evaluate(() =>
    Array.from(document.querySelectorAll(".leaflet-container img")).map(
      (img) => (img as HTMLImageElement).naturalWidth,
    ),
  )

  expect(failedRequests).toEqual([])
  expect(badStatuses).toEqual([])
  expect(widths.length).toBeGreaterThan(0)
  for (const w of widths) {
    expect(w).toBeGreaterThan(0)
  }
})
```

```typescript
test("tapping a household card in the open sheet jumps the wizard", async ({ page }) => {
  // Use a viewport where the sheet top edge visually overlaps the map —
  // mobile portrait is the critical failure mode the volunteer reported.
  await page.setViewportSize({ width: 390, height: 844 }) // iPhone 14 Pro

  // Navigate to canvassing (reuse whatever pattern 108's E2E spec uses).
  await page.goto("/field/<seed-campaign-id>/canvassing")
  await page.waitForSelector('[data-testid="canvassing-map-container"]')

  // Open the All Doors sheet.
  await page.getByRole("button", { name: /all doors/i }).click()
  await page.waitForSelector('[data-testid="door-list-item-1"]')

  // Sanity: the map wrapper is marked inert while the sheet is open.
  const wrapperClass = await page
    .locator('[data-testid="canvassing-map-wrapper"]')
    .getAttribute("class")
  expect(wrapperClass).toContain("canvassing-map-wrapper--inert")

  const ariaHidden = await page
    .locator('[data-testid="canvassing-map-wrapper"]')
    .getAttribute("aria-hidden")
  expect(ariaHidden).toBe("true")

  // Tap the SECOND household card — close enough to the top edge of the
  // sheet that the map is visible behind it.
  await page.getByTestId("door-list-item-2").click()

  // The sheet closes + wizard advances to door 2.
  await page.waitForSelector('[data-testid="canvassing-current-door-copy"]', {
    state: "visible",
  })
  const currentDoor = await page
    .getByTestId("canvassing-current-door-copy")
    .textContent()
  expect(currentDoor).toContain("Door 2 of")
})
```
