function generateShiftData(): Array<{
  name: string; type: string; start_at: string; end_at: string; max_volunteers: number;
}> {
  const shifts = []
  const baseDate = new Date()
  baseDate.setHours(0, 0, 0, 0)

  // 5 shifts this week, morning (9am-12pm)
  for (let d = 0; d < 5; d++) {
    const date = new Date(baseDate)
    date.setDate(date.getDate() + d)
    shifts.push({
      name: `E2E Shift W1 AM ${d + 1}`,
      type: d < 2 ? "canvassing" : d < 4 ? "phone_banking" : "general",
      start_at: new Date(date.setHours(9, 0, 0, 0)).toISOString(),
      end_at: new Date(date.setHours(12, 0, 0, 0)).toISOString(),
      max_volunteers: d < 2 ? 2 : d < 4 ? 5 : 10,
    })
  }
  // 5 shifts this week, afternoon (1pm-5pm)
  for (let d = 0; d < 5; d++) {
    const date = new Date(baseDate)
    date.setDate(date.getDate() + d)
    shifts.push({
      name: `E2E Shift W1 PM ${d + 1}`,
      type: d < 2 ? "canvassing" : d < 4 ? "phone_banking" : "general",
      start_at: new Date(date.setHours(13, 0, 0, 0)).toISOString(),
      end_at: new Date(date.setHours(17, 0, 0, 0)).toISOString(),
      max_volunteers: d < 2 ? 2 : d < 4 ? 5 : 10,
    })
  }
  // 5 shifts next week, morning
  for (let d = 7; d < 12; d++) {
    const date = new Date(baseDate)
    date.setDate(date.getDate() + d)
    shifts.push({
      name: `E2E Shift W2 AM ${d - 6}`,
      type: "general",
      start_at: new Date(date.setHours(9, 0, 0, 0)).toISOString(),
      end_at: new Date(date.setHours(12, 0, 0, 0)).toISOString(),
      max_volunteers: 5,
    })
  }
  // 5 shifts next week, afternoon
  for (let d = 7; d < 12; d++) {
    const date = new Date(baseDate)
    date.setDate(date.getDate() + d)
    shifts.push({
      name: `E2E Shift W2 PM ${d - 6}`,
      type: "general",
      start_at: new Date(date.setHours(13, 0, 0, 0)).toISOString(),
      end_at: new Date(date.setHours(17, 0, 0, 0)).toISOString(),
      max_volunteers: 5,
    })
  }
  return shifts
}
