```tsx
export function RequireOrgRole({ minimum, children, fallback = null }: Props) {
  const { hasOrgRole, isLoading } = useOrgPermissions()
  if (isLoading) return null   // or <></> — do NOT render fallback yet
  if (!hasOrgRole(minimum)) return <>{fallback}</>
  return <>{children}</>
}
```
