```python
@router.get("/campaigns/{campaign_id}/dashboard/...")
async def endpoint(
    campaign_id: uuid.UUID,
    start_date: date | None = Query(default=None),
    end_date: date | None = Query(default=None),
    user: AuthenticatedUser = Depends(require_role("manager")),
    db: AsyncSession = Depends(get_db),
):
    await ensure_user_synced(user, db)
    await set_campaign_context(db, str(campaign_id))
    result = await _service.method(db, campaign_id, start_date, end_date)
    return result
```

```python
@pytest.mark.asyncio
async def test_canvassing_summary_returns_schema():
    session = AsyncMock(spec=AsyncSession)
    # Mock the execute result
    mock_row = MagicMock()
    mock_row._mapping = {
        "doors_knocked": 100,
        "contacts_made": 45,
        "supporter": 20, "undecided": 15, "opposed": 5, "refused": 5,
        "not_home": 30, "moved": 5, "deceased": 2,
        "come_back_later": 10, "inaccessible": 8,
    }
    mock_result = MagicMock()
    mock_result.one.return_value = mock_row
    session.execute.return_value = mock_result

    service = CanvassingDashboardService()
    result = await service.get_summary(session, uuid.uuid4())
    assert isinstance(result, CanvassingSummary)
    assert result.doors_knocked == 100
    assert result.contacts_made == 45
```
